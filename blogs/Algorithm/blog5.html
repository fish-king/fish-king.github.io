计算多边形面积

计算多边形面积可以使用向量外积来计算

struct Point{
    Point(int a,int b):x(a),y(b) {}
    double x,y;
};
double calc_area(vector &p)
{
    double area = 0;
    int len = p.size();
    for(int i = 0;i < len; ++i)
    {
        area += p[i].x * p[(i+1) % len].y - p[i].y * p[(i+1) % len].x; //向量外积计算面积
    }
    return area/2;
}
int main()
{
    int n;
    cin >> n;
    vector p;
    for(int i = 0;i < n; ++i)
    {
        int x,y;
        cin >> x;
        cin >> y;
        p.push_back(Point(x,y));
    }
    cout << calc_area(p);
    return 0;
}
